    Getting Started

    # Installation

    $ git clone git://github.com/tnantoka/konnyaku.git YOUR_BLOG
$ cd YOUR_BLOG/
$ bundle --without test development

    # Database setup

    $ cp config/database.yml.example config/database.yml
$ YOUR_EDITOR config/database.yml
production:
  username: USERNAME
  password: PASSWORD

$ RAILS_ENV=production rake db:drop db:create db:migrate db:seed

    # Configuration

    $ cp config/application.yml.example config/application.yml
$ YOUR_EDITOR config/application.yml

    # Run

    $ rake assets:precompile
$ rails s -e production
